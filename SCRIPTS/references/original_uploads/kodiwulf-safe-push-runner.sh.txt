#!/usr/bin/env bash

set -Eeuo pipefail

GITHUB_USER="n-e-o-w-u-l-f"
REPO_NAME="kodiwulf-repo"
TARGET_DIR="$HOME/Projekte/${GITHUB_USER}/${REPO_NAME}"
REMOTE_URL="https://github.com/${GITHUB_USER}/${REPO_NAME}.git"

echo "============================================================"
echo "KodiWulf Safe Push Runner"
echo "============================================================"
echo "Zeit:     $(date)"
echo "Zielrepo: $TARGET_DIR"
echo "Remote:   $REMOTE_URL"
echo

echo "============================================================"
echo "Prüfe Zielrepo"
echo "============================================================"

if [ ! -d "$TARGET_DIR" ]; then
  echo "[FEHLER] Zielrepo existiert nicht:"
  echo "$TARGET_DIR"
  exit 10
fi

cd "$TARGET_DIR"

if ! git rev-parse --is-inside-work-tree >/dev/null 2>&1; then
  echo "[FEHLER] Zielordner ist kein Git-Repository."
  exit 11
fi

echo "[OK] Zielrepo ist ein Git-Repository."
echo

echo "============================================================"
echo "Git-Status"
echo "============================================================"
git status --short
echo

if [ -n "$(git status --short)" ]; then
  echo "[ABBRUCH] Arbeitsbaum ist nicht sauber."
  echo "[INFO] Es wird nichts gepusht."
  exit 12
fi

echo "[OK] Arbeitsbaum ist sauber."
echo

echo "============================================================"
echo "Remote prüfen"
echo "============================================================"

if git remote get-url origin >/dev/null 2>&1; then
  CURRENT_REMOTE="$(git remote get-url origin)"
  echo "[INFO] Aktuelle origin:"
  echo "$CURRENT_REMOTE"

  if [ "$CURRENT_REMOTE" != "$REMOTE_URL" ]; then
    echo "[INFO] Korrigiere origin auf:"
    echo "$REMOTE_URL"
    git remote set-url origin "$REMOTE_URL"
  fi
else
  echo "[INFO] Origin fehlt. Lege origin an."
  git remote add origin "$REMOTE_URL"
fi

git remote -v
echo

echo "============================================================"
echo "Branch prüfen"
echo "============================================================"

CURRENT_BRANCH="$(git branch --show-current)"
echo "[INFO] Aktiver Branch: $CURRENT_BRANCH"

if [ "$CURRENT_BRANCH" != "main" ]; then
  echo "[INFO] Wechsle zu main."
  git checkout main
fi

git branch -vv
echo

echo "============================================================"
echo "Letzte Commits"
echo "============================================================"
git log --oneline --decorate --graph -5
echo

echo "============================================================"
echo "Push main -> origin/main"
echo "============================================================"
echo "[WICHTIG] Username: n-e-o-w-u-l-f"
echo "[WICHTIG] Password: Personal Access Token, NICHT GitHub-Passwort"
echo "[HINWEIS] Beim Einfügen des Tokens sieht man meistens nichts. Das ist normal."
echo

git push -u origin main

echo
echo "============================================================"
echo "Status nach Push"
echo "============================================================"
git branch -vv
echo
git status
echo

echo "============================================================"
echo "Fertig"
echo "============================================================"
echo "[OK] Script wurde beendet."
